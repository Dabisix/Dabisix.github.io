package gold;

import java.io.BufferedReader;
import java.io.IOException;
import java.io.InputStreamReader;
import java.util.LinkedList;
import java.util.Queue;
import java.util.StringTokenizer;

public class G2250_treesHW {
	static int N;
	static int[][] node;
	static int[] cols;

	static int dfs(int now, int cnt) {
		if(now==-1) return cnt;         // ì¡´ì¬í•˜ì§€ ì•ŠëŠ” ë…¸ë“œ pass

		int c = dfs(node[now][0], cnt); // 1. ì™¼ìª½ ì„œë¸ŒíŠ¸ë¦¬ ì—´ ë²ˆí˜¸ ë¶€ì—¬
		cols[now] = c;					        // 2. í˜„ì¬ ë…¸ë“œ ì—´ ë²ˆí˜¸ ë¶€ì—¬
		return dfs(node[now][1], ++c);	// 3. ì˜¤ë¥¸ìª½ ì„œë¸ŒíŠ¸ë¦¬ ì—´ ë²ˆí˜¸ ë¶€ì—¬
	}

	static String bfs(int startNode) {
		Queue<Integer> queue = new LinkedList<Integer>();
		queue.add(startNode);
		int max_width=-1, max_level=-1, now_level=0;

		while(!queue.isEmpty()) {
			now_level++;

			Object[] tmp = queue.toArray(); // 1. í˜„ì¬ ë ˆë²¨ ë„ˆë¹„ ê³„ì‚°
			int now_width = cols[(int)tmp[tmp.length-1]] - cols[(int)tmp[0]] + 1;
			if(max_width < now_width) {
				max_width = now_width;
				max_level = now_level;
			}

			int size = queue.size();	    	// 2. ë‹¤ìŒ ë ˆë²¨ ë…¸ë“œ ìˆ˜ì§‘
			for(int i=0; i<size; i++) {
				int now = queue.poll();
				if(node[now][0] != -1) queue.add(node[now][0]);
				if(node[now][1] != -1) queue.add(node[now][1]);
			}
		}
		return max_level + " " + max_width;
	}

	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		node = new int[N+1][2];
		cols = new int[N+1];

		// 1. ë…¸ë“œ ì €ì¥
		boolean[] isChild = new boolean[N+1];
		for(int i=0; i<N; i++) {
			StringTokenizer st = new StringTokenizer(br.readLine());
			int now = Integer.parseInt(st.nextToken());
			for(int j=0; j<2; j++) {
				node[now][j] = Integer.parseInt(st.nextToken());
				if(node[now][j] != -1) isChild[node[now][j]] = true;
			}
		}

		// 2. ë£¨íŠ¸ë…¸ë“œ ì°¾ê¸°
		int root = -1;
		for(int i=1; i<=N; i++) {
			if(!isChild[i]) {
				root = i;
				break;
			}
		}

		// 3. dfsë¡œ ìˆœì„œëŒ€ë¡œ ì—´ ë²ˆí˜¸ ì£¼ê¸°
		dfs(root, 1);

		// 4. bfsë¡œ ë ˆë²¨ ë³„ ë„ˆë¹„ êµ¬í•˜ê¸°
		System.out.println(bfs(root));
	}

}
